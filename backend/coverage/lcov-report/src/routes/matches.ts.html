        relations: ['ai1', 'ai2', 'threads']
      });
    }
 
    const formattedMatches = matches.map(match => {
      if (isAI) {
        // For AI view, show the match as-is
        return {
          id: match.id,
          threadId: match.threads?.[0]?.id ?? null,
          threads: match.threads,
          ai1: match.ai1,
          ai2: match.ai2
        };
      } else {
        // For user view, show the partner AI
        const aiProfile = match.ai1.user?.id === userId ? match.ai2 : match.ai1;
        return {
          id: match.id,
          threadId: match.threads?.[0]?.id ?? null,
          threads: match.threads,
          user: { id: user.id, username: user.name },
          aiProfile,
          ai1: match.ai1,
          ai2: match.ai2
        };
      }
    });
